18bool Application::Initialize()
19{
20 if (!D3DApp::Initialize())
21 return false;
22
23 mCamera.SetPosition(0, 7, 0);
24 mCamera.Pitch(3.14 / 2);
25
26 // Reset the command list to prep for initialization commands.
27 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
28
29 // Get the increment size of a descriptor in this heap type. This is hardware specific,
30 // so we have to query this information.
31 mCbvSrvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
32
33 LoadTextures();
34 BuildRootSignature();
35 BuildDescriptorHeaps();
36 BuildShadersAndInputLayout();
37 BuildShapeGeometry();
38 BuildMaterials();
39 //BuildRenderItems();
40
41 mStateStack = StateStack(State::Context(this, &mPlayer));
42 RegisterStates();
43
44 BuildFrameResources();
45 BuildPSOs();
46 mStateStack.pushState(States::Title);
47 // Execute the initialization commands.
48 ThrowIfFailed(mCommandList->Close());
49 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
50 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
51
52 // Wait until initialization is complete.
53 FlushCommandQueue();
54
55 return true;
56}

78 // Cycle through the circular frame resource array.
79 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
80 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
81
82 // Has the GPU finished processing the commands of the current frame resource?
83 // If not, wait until the GPU has completed commands up to this fence point.
84 if (mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
85 {
86 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
87 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
88 WaitForSingleObject(eventHandle, INFINITE);
89 CloseHandle(eventHandle);
90 }
91
92 AnimateMaterials(gt);
93 UpdateObjectCBs(gt);
94 UpdateMaterialBuffer(gt);
95 UpdateMainPassCB(gt);

269void Application::UpdateMainPassCB(const GameTimer& gt)
270{
271 XMMATRIX view = mCamera.GetView();
272 XMMATRIX proj = mCamera.GetProj();
273
274 auto detView = XMMatrixDeterminant(view);
275 auto detProj = XMMatrixDeterminant(proj);
276 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
277 auto detViewProj = XMMatrixDeterminant(viewProj);
278
279 XMMATRIX invView = XMMatrixInverse(&detView, view);
280 XMMATRIX invProj = XMMatrixInverse(&detProj, proj);
281 XMMATRIX invViewProj = XMMatrixInverse(&detViewProj, viewProj);
282
283 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
284 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
285 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
286 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
287 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
288 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
289 mMainPassCB.EyePosW = mCamera.GetPosition3f();
290 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
291 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
292 mMainPassCB.NearZ = 1.0f;
293 mMainPassCB.FarZ = 1000.0f;
294 mMainPassCB.TotalTime = gt.TotalTime();
295 mMainPassCB.DeltaTime = gt.DeltaTime();
296 mMainPassCB.AmbientLight = { 0.25f, 0.25f, 0.35f, 1.0f };
297 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
298 mMainPassCB.Lights[0].Strength = { 0.8f, 0.8f, 0.8f };
299 mMainPassCB.Lights[1].Direction = { -0.57735f, -0.57735f, 0.57735f };
300 mMainPassCB.Lights[1].Strength = { 0.4f, 0.4f, 0.4f };
301 mMainPassCB.Lights[2].Direction = { 0.0f, -0.707f, -0.707f };
302 mMainPassCB.Lights[2].Strength = { 0.2f, 0.2f, 0.2f };
303
304 auto currPassCB = mCurrFrameResource->PassCB.get();

426void Application::BuildRootSignature()
427{
428 // 5 slots with two descriptor tables, one structured buffer SRV, and two CBVs.
429 CD3DX12_DESCRIPTOR_RANGE texTable0;
430 texTable0.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0, 0); // sky cube map
431
432 CD3DX12_DESCRIPTOR_RANGE texTable1;
433 texTable1.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 17, 1, 0); // all scene textures
434
435 CD3DX12_ROOT_PARAMETER slotRootParameter[5];
436
437 slotRootParameter[0].InitAsConstantBufferView(0); // ObjectCB
438 slotRootParameter[1].InitAsConstantBufferView(1); // PassCB
439 slotRootParameter[2].InitAsShaderResourceView(0, 1); // MaterialBuffer (structured)
440 slotRootParameter[3].InitAsDescriptorTable(1, &texTable0, D3D12_SHADER_VISIBILITY_PIXEL); // sky
441 slotRootParameter[4].InitAsDescriptorTable(1, &texTable1, D3D12_SHADER_VISIBILITY_PIXEL); // textures
442
443 auto staticSamplers = GetStaticSamplers();
444
445 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(5, slotRootParameter,
446 (UINT)staticSamplers.size(), staticSamplers.data(),
447 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
448
449 ComPtr<ID3DBlob> serializedRootSig = nullptr;
450 ComPtr<ID3DBlob> errorBlob = nullptr;
451 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
452 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
453
454 if (errorBlob != nullptr)
455 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
456
457 ThrowIfFailed(hr);
458
459 ThrowIfFailed(md3dDevice->CreateRootSignature(
460 0,
461 serializedRootSig->GetBufferPointer(),
462 serializedRootSig->GetBufferSize(),
463 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
464}

542void Application::BuildShapeGeometry()
543{
544 GeometryGenerator geoGen;
545 GeometryGenerator::MeshData box = geoGen.CreateBox(1.0f, 1.0f, 1.0f, 3);
546 GeometryGenerator::MeshData grid = geoGen.CreateGrid(20.0f, 30.0f, 60, 40);
547 GeometryGenerator::MeshData sphere = geoGen.CreateSphere(0.5f, 20, 20);
548 GeometryGenerator::MeshData cylinder = geoGen.CreateCylinder(0.5f, 0.3f, 3.0f, 20, 20);
549 GeometryGenerator::MeshData quad = geoGen.CreateQuad(0.0f, 0.0f, 1.0f, 1.0f, 0.0f);
550
551 UINT boxVertexOffset = 0;
552 UINT gridVertexOffset = (UINT)box.Vertices.size();
553 UINT sphereVertexOffset = gridVertexOffset + (UINT)grid.Vertices.size();
554 UINT cylinderVertexOffset = sphereVertexOffset + (UINT)sphere.Vertices.size();
555 UINT quadVertexOffset = cylinderVertexOffset + (UINT)cylinder.Vertices.size();
556
557 UINT boxIndexOffset = 0;
558 UINT gridIndexOffset = (UINT)box.Indices32.size();
559 UINT sphereIndexOffset = gridIndexOffset + (UINT)grid.Indices32.size();
560 UINT cylinderIndexOffset = sphereIndexOffset + (UINT)sphere.Indices32.size();
561 UINT quadIndexOffset = cylinderIndexOffset + (UINT)cylinder.Indices32.size();
562
563 SubmeshGeometry boxSubmesh;
564 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
565 boxSubmesh.StartIndexLocation = boxIndexOffset;
566 boxSubmesh.BaseVertexLocation = boxVertexOffset;
567

569 gridSubmesh.IndexCount = (UINT)grid.Indices32.size();
570 gridSubmesh.StartIndexLocation = gridIndexOffset;
571 gridSubmesh.BaseVertexLocation = gridVertexOffset;

573 SubmeshGeometry sphereSubmesh;
574 sphereSubmesh.IndexCount = (UINT)sphere.Indices32.size();
575 sphereSubmesh.StartIndexLocation = sphereIndexOffset;

578 SubmeshGeometry cylinderSubmesh;
579 cylinderSubmesh.IndexCount = (UINT)cylinder.Indices32.size();
580 cylinderSubmesh.StartIndexLocation = cylinderIndexOffset;
581 cylinderSubmesh.BaseVertexLocation = cylinderVertexOffset;

583 SubmeshGeometry quadSubmesh;
584 quadSubmesh.IndexCount = (UINT)quad.Indices32.size();
585 quadSubmesh.StartIndexLocation = quadIndexOffset;
586 quadSubmesh.BaseVertexLocation = quadVertexOffset;
587
588 auto totalVertexCount =
589 box.Vertices.size() +
590 grid.Vertices.size() +
591 sphere.Vertices.size() +
592 cylinder.Vertices.size() +
593 quad.Vertices.size();
594
595 std::vector<Vertex> vertices(totalVertexCount);
596
597 UINT k = 0;
598 for (size_t i = 0; i < box.Vertices.size(); ++i, ++k)
599 {
600 vertices[k].Pos = box.Vertices[i].Position;
601 vertices[k].Normal = box.Vertices[i].Normal;
602 vertices[k].TexC = box.Vertices[i].TexC;
603 vertices[k].TangentU = box.Vertices[i].TangentU;
604 }
605 for (size_t i = 0; i < grid.Vertices.size(); ++i, ++k)
606 {
607 vertices[k].Pos = grid.Vertices[i].Position;
608 vertices[k].Normal = grid.Vertices[i].Normal;
609 vertices[k].TexC = grid.Vertices[i].TexC;
610 vertices[k].TangentU = grid.Vertices[i].TangentU;
611 }
612 for (size_t i = 0; i < sphere.Vertices.size(); ++i, ++k)
613 {
614 vertices[k].Pos = sphere.Vertices[i].Position;
615 vertices[k].Normal = sphere.Vertices[i].Normal;
616 vertices[k].TexC = sphere.Vertices[i].TexC;
617 vertices[k].TangentU = sphere.Vertices[i].TangentU;
618 }
619 for (size_t i = 0; i < cylinder.Vertices.size(); ++i, ++k)
620 {
621 vertices[k].Pos = cylinder.Vertices[i].Position;
622 vertices[k].Normal = cylinder.Vertices[i].Normal;
623 vertices[k].TexC = cylinder.Vertices[i].TexC;
624 vertices[k].TangentU = cylinder.Vertices[i].TangentU;
625 }
626 for (size_t i = 0; i < quad.Vertices.size(); ++i, ++k)
627 {
628 vertices[k].Pos = quad.Vertices[i].Position;
629 vertices[k].Normal = quad.Vertices[i].Normal;
630 vertices[k].TexC = quad.Vertices[i].TexC;
631 vertices[k].TangentU = quad.Vertices[i].TangentU;
632 }
633
634 std::vector<std::uint16_t> indices;
635 indices.insert(indices.end(), std::begin(box.GetIndices16()), std::end(box.GetIndices16()));
636 indices.insert(indices.end(), std::begin(grid.GetIndices16()), std::end(grid.GetIndices16()));
637 indices.insert(indices.end(), std::begin(sphere.GetIndices16()), std::end(sphere.GetIndices16()));
638 indices.insert(indices.end(), std::begin(cylinder.GetIndices16()), std::end(cylinder.GetIndices16()));
639 indices.insert(indices.end(), std::begin(quad.GetIndices16()), std::end(quad.GetIndices16()));
640
641 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
642 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
643
644 auto geo = std::make_unique<MeshGeometry>();
645 geo->Name = "shapeGeo";
646
647 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
648 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
649
650 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
651 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
652
653 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
654 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
655
656 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
657 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
658
659 geo->VertexByteStride = sizeof(Vertex);
660 geo->VertexBufferByteSize = vbByteSize;
661 geo->IndexFormat = DXGI_FORMAT_R16_UINT;
662 geo->IndexBufferByteSize = ibByteSize;
663
664 geo->DrawArgs["box"] = boxSubmesh;
665 geo->DrawArgs["grid"] = gridSubmesh;
666 geo->DrawArgs["sphere"] = sphereSubmesh;
667 geo->DrawArgs["cylinder"] = cylinderSubmesh;
668 geo->DrawArgs["quad"] = quadSubmesh;
669
670 mGeometries[geo->Name] = std::move(geo);
671}
